### 点击事件，传入一个元素，然后控制浏览器点击他，比如传入一个button会让浏览器点击这个button，
### 代替我们鼠标直接操作
def oneclick(temp,timesleep=5):# ,timesleep=5
    webdriver.ActionChains(driver).move_to_element(temp).click(temp).perform()
    time.sleep(timesleep)

def huadong(times=3):
    """
    对于一个话题有个坑就是，如果你不把浏览器滑到底，他是不会出现加载更多评论这个a标签的，所以这个函数是模拟
    将屏幕滑到底，一共需要滑动n次，这也是为什么用浏览器控制会更好的原因。
    """
    for i in range(times):
        driver.execute_script("window.scrollTo(0,document.body.scrollHeight);")
        time.sleep(0.5)

def loadAllData(timesleep=1):
    huadong(4)## 滑动到页面底部，会出现加载更多的a标签，这个a标签的action-type属性是唯一的
    flag = True
    time1 = time.time()
    count = 0
    while flag: ### 点完所有的更多的评论，知道没有更多的评论可以加载为止
        try:
            count+=1
            time.sleep(1)
            temp = driver.find_element_by_css_selector('[action-type="click_more_comment"]')
#             temp.click()
            oneclick(temp, 0.1)
        except Exception as e:
#             print(e)
            flag = False
    print(f"all is {count}, and spend {time.time()-time1}...")

def computeNums():
    """
    获取评论的一些信息，包括，转发数，评论数，点赞数
    """
    temp = driver.find_element_by_css_selector("[class='WB_row_line WB_row_r4 clearfix S_line2']")
    temp = temp.text.split('\n')[1:]
    temp = [each[1:] for each in temp]
    return temp


# In[9]:


def computeAll():
    """
    获取发这个话题的机构的名字，时间、来源，和主要内容
    """
    temp = driver.find_element_by_css_selector("[class='WB_feed_detail clearfix']")

    temp = temp.find_elements_by_class_name('WB_detail')[0].text.split('\n')[:-4]
    if "来自" in temp[1]:
        time1,frm = temp[1].split('来自')
    else:
        time1,frm  = temp[1],""
    return temp[0], time1.strip().replace(" ","_").replace(":","&"), frm.strip().replace(" ","_").replace(":","&"), ''.join(temp[2:])

def computeCon():
    """
    获取这个页面所有的一级评论
    """
    content = driver.find_elements_by_class_name('list_con')
    names = []
    neirong = []
    dianzanshu = []
    shijian = []
    # result = [each.text.split('\n')[:-1] for each  in content]

    for each in tqdm(content):
        temp = each.text.split('\n')
        names.append(temp[0].split('：')[0])
        neirong.append(temp[0].split('：')[1])    
        dianzanshu.append(temp[2][1:])    
        shijian.append(temp[3])
    
    data = pd.DataFrame(columns=['names', 'neirong', 'dianzanshu', 'shijian'])
    data['names'] =names
    data['neirong'] =neirong    
    data['dianzanshu'] =dianzanshu    
    data['shijian'] =shijian    
    return data

def saveOneresult(basePath):
    """
    存储一个话题的所有信息以及所有评论
    """
    loadAllData(2)

    nums = computeNums()

    name,times,frm,maincontent = computeAll()
    storeName = times+"_"+name+"_"+frm
    reply = computeCon()
#     print(maincontent,name,nums,times)
    print(storeName) ## 存储名字以时间_名字_来源命令
    with open(basePath+'all.csv', 'a',encoding='utf8') as f:
        ### 存储这条话题基本的信息在all.csv 里面
        f.write(name+','+",".join(times)+","+','.join(nums)+',' +maincontent+'\n')
    reply.to_csv(basePath+storeName+'.csv', index=None)

def oneDaySave(one_day):
    """
    传入一天的时间
    然后浏览器访问这一天的热门话题的搜索结果页面
    """
    url = f'https://s.weibo.com/weibo/%E7%96%AB%E6%83%85?q=%E7%96%AB%E6%83%85&xsort=hot&suball=1&timescope=custom:{one_day}-0:{one_day}-23&Refer=g'
    driver.get(url)
    print(url)
    for index in range(5):
        mustFlag = True
        counts = 0
        while mustFlag:
            try:
                each = driver.find_element_by_xpath(f'//*[@id="pl_feedlist_index"]/div[2]/div[{index+1}]')
                aHerf = each.find_elements_by_xpath(f'./div/div[2]/ul/li[3]/a')
                ### aHerf 是 话题下面评论的那个a标签，只有点击了评论，评论才会被加载出来
                oneclick(aHerf[0])
                each = driver.find_element_by_xpath(f'//*[@id="pl_feedlist_index"]/div[2]/div[{index+1}]')
                temp = each.find_elements_by_class_name('card-more-a') ### 一般评论够多的话才会有加载更多评论
                ### 的选项，所以此处做一个筛选，如果评论少于4条是不会有这个的，如果少于4条还不如不要
                if len(temp):

                    temp1 = temp[0].find_elements_by_tag_name('a')
                    print(index, temp1[0].text)
                    ### 进入一个具体的话题或者文章的具体页面
                    oneclick(temp[0])
                    ### 获取这个页面的评论
                    saveOneresult(basePath)
                    driver.back()
                else:
                    print(index,  0)
                time.sleep(0.1)
                mustFlag = False
            except Exception as e:
                ### 有时候微博还是会把你判断为爬虫，所以有个出错控制，这个时候只需要刷新浏览器即可
                ### 但是刷新次数过多也不好
                counts+=1
                print(e)
                with open(basePath+'sunhuai.csv', 'a',encoding='utf8') as f:
                    f.write(one_day+','+str(index)+','+str(e)+"\n")
                if counts==2:
                    mustFlag= False
                elif counts ==1:
                    driver.refresh()
